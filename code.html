Encryption/Decryption

This code segment uses conditionals and loops to encrypt and decrypt messages.

answer = input("(E) Encrypt Message\n(D) Decrypt Message\n(Q) Quit\nEnter your choice: ")
while (answer != "Q"):
  message = input("Enter the message: ")
  value = input("Enter the shift value: ")
  if (answer == "E"):
    ans = ""
    for i in range(len(message)):
      ch = message[i]
      if ch==" ":
        ans+=" "
      elif ch.isupper():
        ans += chr((ord(ch) + int(value) - 65) % 26 + 65)
      else:
        ans += chr((ord(ch) + int(value) - 97) % 26 + 97)
    print("Encrypted message: " + ans)
  elif (answer == "D"):
    ans = ""
    for i in range(len(message)):
      ch = message[i]
      if ch==" ":
        ans+=" "
      elif ch.isupper():
        ans += chr((ord(ch) - int(value) - 65) % 26 + 65)
      else:
        ans += chr((ord(ch) - int(value) - 97) % 26 + 97)
    print("Decrypted message: " + ans)
  answer = input("(E) Encrypt Message\n(D) Decrypt Message\n(Q) Quit\nEnter your choice: ")
